# ── DeepFilterNet2 — Sprint C: 7th competing algorithm ───────────────
#  A 48 kHz single-channel neural denoiser (Schröter et al., ICASSP 2022,
#  arXiv:2205.05474) run on the reference mic. It competes against the 6
#  beamformers so the leaderboard answers "does 8-mic beamforming beat
#  1-mic + neural denoising?". It is NOT a winner-polish step.
#
#  NOTE: the only package on PyPI is `deepfilternet` 0.5.6, which bundles
#  the DeepFilterNet2 / DeepFilterNet3 pretrained weights and exposes the
#  init_df()/enhance() API. OCTOVOX pins DFN to CPU (CUDA_VISIBLE_DEVICES=""
#  + config DEVICE=cpu) to dodge the historical CUDA/CPU tensor-mismatch
#  crash; see octovox_app/services/pipeline.py and tests/pilot_dfn2.py.
deepfilternet==0.5.6
soundfile>=0.12            # WAV I/O used by the DFN pilot

# ── Neural-MVDR-WPE — WPE dereverb + Silero VAD + MVDR ───────────────

# ── torch (shared by DeepFilterNet2, Neural-MVDR-WPE, Silero VAD) ────
#  CPU build is sufficient — OCTOVOX pins DFN to CPU regardless. For GPU
#  acceleration of Silero VAD only, install the matching CUDA build per
#  the notes in requirements.txt instead.
